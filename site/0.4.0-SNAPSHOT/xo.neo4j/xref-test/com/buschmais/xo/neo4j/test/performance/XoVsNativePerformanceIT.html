182     private <Entity, Relation> List<Measurement> run(ApiUnderTest<Entity, Relation> apiUnderTest) {
183         LOGGER.info("Starting run with API '" + apiUnderTest + "'");
184         List<Measurement> measurements = new ArrayList<>(NUMBER_OF_RUNS);
185         for (int i = 0; i < NUMBER_OF_RUNS; i++) {
186             dropDatabase();
187             apiUnderTest.begin();
188             long start = System.currentTimeMillis();
189             Entity root = apiUnderTest.createEntity();
190             apiUnderTest.setName(root, "1");
191 
192             long counter = 1;
193             counter += addChildren(apiUnderTest, root, 2, "1");
194             apiUnderTest.commit();
195             long stop = System.currentTimeMillis();
196 
197             Measurement measurement = new Measurement(counter, start, stop);
198             LOGGER.info("counter=" + measurement.getCounter() + ", time=" + measurement.getDuration() + "ms" + ", speed=" + measurement.getSpeed() + " vertices/s");
199             measurements.add(measurement);
200         }
201         LOGGER.info("Finished run with API " + apiUnderTest);
202         return measurements;
203     }

205     private <Entity, Relation> long addChildren(ApiUnderTest<Entity, Relation> apiUnderTest, Entity parent, int i, String namePrefix) {
206         if (i > TREE_DEPTH) {
207             return 0;
208         }
209         long counter = 0;
210         for (int id = 1; id <= i; id++) {
211             String name = namePrefix + id;
212             Entity child = apiUnderTest.createEntity();
213             apiUnderTest.setName(child, name);
214             apiUnderTest.createRelation(parent, child);
215             counter++;
216             counter += addChildren(apiUnderTest, child, i + 1, name);
217         }
218         return counter;
219     }

233     private void print(List<Measurement> measurements) {
234         System.out.println("Counter\tDuration [ms]\tSpeed [vertices/s]");
235         System.out.println("--------------------------------------------------");
236         long durationSum = 0;
237         double speedSum = 0.0;
238         for (Measurement measurement : measurements) {
239             long counter = measurement.getCounter();
240             long duration = measurement.getDuration();
241             double speed = measurement.getSpeed();
242             String format = MessageFormat.format("{0}\t{1}\t{2,number,#.##}",
243                     counter, duration, speed);
244             System.out.println(format);
245             speedSum += speed;
246             durationSum += duration;
247         }
248         System.out
249                 .println("--------------------------------------------------");
250         double durationAvg = (double) durationSum

252         System.out.println(MessageFormat.format(
