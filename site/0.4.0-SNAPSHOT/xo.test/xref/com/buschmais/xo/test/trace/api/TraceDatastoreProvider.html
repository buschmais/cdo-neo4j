21  public class TraceDatastoreProvider<EntityMetadata extends DatastoreEntityMetadata<EntityDiscriminator>, EntityDiscriminator, RelationMetadata extends DatastoreRelationMetadata<RelationDiscriminator>, RelationDiscriminator> implements XODatastoreProvider<EntityMetadata, EntityDiscriminator, RelationMetadata, RelationDiscriminator> {
22  
23      public static final String DELEGATE_KEY = TraceDatastoreProvider.class.getPackage().getName() + ".DelegateProvider";
24  
25      @Override
26      public Datastore<?, EntityMetadata, EntityDiscriminator, RelationMetadata, RelationDiscriminator> createDatastore(XOUnit xoUnit) {
27          Properties properties = xoUnit.getProperties();
28          String delegateProviderName = properties.getProperty(DELEGATE_KEY);
29          if (delegateProviderName == null) {
30              throw new XOException("Property '" + DELEGATE_KEY + "' must be specified using the class name of a data store provider.");
31          }
32          Class<XODatastoreProvider> delegateProviderType = ClassHelper.getType(delegateProviderName);
33          XODatastoreProvider<EntityMetadata, EntityDiscriminator, RelationMetadata, RelationDiscriminator> delegateProvider = ClassHelper.newInstance(delegateProviderType);
34          Datastore<?, EntityMetadata, EntityDiscriminator, RelationMetadata, RelationDiscriminator> delegateDatastore = delegateProvider.createDatastore(xoUnit);
35          TraceMonitor traceMonitor = new TraceMonitor(xoUnit);
36          InterceptorFactory interceptorFactory = new InterceptorFactory(Arrays.asList(new TraceMonitorInterceptor(traceMonitor)));
37          return new TraceDatastore<>(interceptorFactory.addInterceptor(delegateDatastore, Datastore.class), interceptorFactory, traceMonitor);
38      }
