100     private <T> Map<PrimitivePropertyMethodMetadata<PropertyMetadata>, Object> prepareExample(Example<T> example, Class<?> type, Class<?>... types) {
101         Map<PrimitivePropertyMethodMetadata<PropertyMetadata>, Object> exampleEntity = new HashMap<>();
102         InstanceInvocationHandler invocationHandler = new InstanceInvocationHandler(exampleEntity, new ExampleProxyMethodService(type, sessionContext));
103         List<Class<?>> effectiveTypes = new ArrayList<>();
104         effectiveTypes.add(type);
105         effectiveTypes.addAll(Arrays.asList(types));
106         T instance = sessionContext.getProxyFactory().createInstance(invocationHandler, effectiveTypes.toArray(new Class<?>[effectiveTypes.size()]),
107                 CompositeObject.class);
108         example.prepare(instance);
109         return exampleEntity;
110     }

120     private <T> ResultIterable<T> findByExample(Class<?> type, Map<PrimitivePropertyMethodMetadata<PropertyMetadata>, Object> entity) {
121         EntityTypeMetadata<EntityMetadata> entityTypeMetadata = sessionContext.getMetadataProvider().getEntityMetadata(type);
122         EntityDiscriminator entityDiscriminator = entityTypeMetadata.getDatastoreMetadata().getDiscriminator();
123         if (entityDiscriminator == null) {
124             throw new XOException("Type " + type.getName() + " has no discriminator (i.e. cannot be identified in datastore).");
125         }
126         ResultIterator<Entity> iterator = sessionContext.getDatastoreSession().getDatastoreEntityManager()
127                 .findEntity(entityTypeMetadata, entityDiscriminator, entity);
128         ResultIterator<T> resultIterator = new ResultIterator<T>() {
129 
130             @Override
131             public boolean hasNext() {
132                 return iterator.hasNext();
133             }
134 
135             @Override
136             public T next() {
137                 Entity entity = iterator.next();
138                 AbstractInstanceManager<EntityId, Entity> entityInstanceManager = sessionContext.getEntityInstanceManager();
139                 return entityInstanceManager.readInstance(entity);
140             }
141 
142             @Override
143             public void remove() {
144                 throw new UnsupportedOperationException("Cannot remove instance.");
145             }
146 
147             @Override
148             public void close() {
149                 iterator.close();
150             }
151         };
152         XOTransaction xoTransaction = sessionContext.getXOTransaction();
153         final ResultIterator<T> transactionalIterator = xoTransaction != null ? new TransactionalResultIterator<>(resultIterator, xoTransaction) : resultIterator;
154         return sessionContext.getInterceptorFactory().addInterceptor(new AbstractResultIterable<T>() {
155             @Override
156             public ResultIterator<T> iterator() {
157                 return transactionalIterator;
158             }
159         }, ResultIterable.class);

191     private CompositeObject createByExample(Class<?> type, Class<?>[] types, Map<PrimitivePropertyMethodMetadata<PropertyMetadata>, Object> exampleEntity) {
192         TypeMetadataSet<EntityTypeMetadata<EntityMetadata>> effectiveTypes = getEffectiveTypes(type, types);
193         Set<EntityDiscriminator> entityDiscriminators = sessionContext.getMetadataProvider().getEntityDiscriminators(effectiveTypes);
194         DatastoreSession<EntityId, Entity, EntityMetadata, EntityDiscriminator, RelationId, Relation, RelationMetadata, RelationDiscriminator, PropertyMetadata> datastoreSession = sessionContext
195                 .getDatastoreSession();
196         Entity entity = datastoreSession.getDatastoreEntityManager().createEntity(effectiveTypes, entityDiscriminators, exampleEntity);
197         AbstractInstanceManager<EntityId, Entity> entityInstanceManager = sessionContext.getEntityInstanceManager();
198         CompositeObject instance = entityInstanceManager.createInstance(entity);
199         sessionContext.getInstanceListenerService().postCreate(instance);
200         return instance;

216         AbstractInstanceManager<EntityId, Entity> entityInstanceManager = sessionContext.getEntityInstanceManager();
217         AbstractInstanceManager<RelationId, Relation> relationInstanceManager = sessionContext.getRelationInstanceManager();
218         if (entityInstanceManager.isInstance(instance)) {

229     public <T, M> CompositeObject migrate(T instance, MigrationStrategy<T, M> migrationStrategy, Class<M> targetType, Class<?>... targetTypes) {
230         AbstractInstanceManager<EntityId, Entity> entityInstanceManager = sessionContext.getEntityInstanceManager();
231         Entity entity = entityInstanceManager.getDatastoreType(instance);
232         DatastoreSession<EntityId, Entity, EntityMetadata, EntityDiscriminator, RelationId, Relation, RelationMetadata, RelationDiscriminator, PropertyMetadata> datastoreSession = sessionContext
233                 .getDatastoreSession();
234         Set<EntityDiscriminator> entityDiscriminators = datastoreSession.getDatastoreEntityManager().getEntityDiscriminators(entity);
235         MetadataProvider<EntityMetadata, EntityDiscriminator, RelationMetadata, RelationDiscriminator> metadataProvider = sessionContext.getMetadataProvider();
236         TypeMetadataSet<EntityTypeMetadata<EntityMetadata>> types = metadataProvider.getTypes(entityDiscriminators);
237         TypeMetadataSet<EntityTypeMetadata<EntityMetadata>> effectiveTargetTypes = getEffectiveTypes(targetType, targetTypes);
238         Set<EntityDiscriminator> targetEntityDiscriminators = metadataProvider.getEntityDiscriminators(effectiveTargetTypes);
239         datastoreSession.getDatastoreEntityManager().migrateEntity(entity, types, entityDiscriminators, effectiveTargetTypes, targetEntityDiscriminators);
240         entityInstanceManager.removeInstance(instance);
241         CompositeObject migratedInstance = entityInstanceManager.updateInstance(entity);
242         if (migrationStrategy != null) {
243             migrationStrategy.migrate(instance, migratedInstance.as(targetType));
244         }
245         entityInstanceManager.closeInstance(instance);
246         return migratedInstance;
247     }

264     @Override
265     public <T> void delete(T instance) {
266         AbstractInstanceManager<EntityId, Entity> entityInstanceManager = sessionContext.getEntityInstanceManager();
267         AbstractInstanceManager<RelationId, Relation> relationInstanceManager = sessionContext.getRelationInstanceManager();
268         DatastoreSession<EntityId, Entity, EntityMetadata, EntityDiscriminator, RelationId, Relation, RelationMetadata, RelationDiscriminator, PropertyMetadata> datastoreSession = sessionContext
269                 .getDatastoreSession();
270         if (entityInstanceManager.isInstance(instance)) {
271             Entity entity = entityInstanceManager.getDatastoreType(instance);
272             sessionContext.getInstanceListenerService().preDelete(instance);
273             datastoreSession.getDatastoreEntityManager().deleteEntity(entity);
274             entityInstanceManager.removeInstance(instance);
275             entityInstanceManager.closeInstance(instance);
276             sessionContext.getInstanceListenerService().postDelete(instance);
277         } else if (relationInstanceManager.isInstance(instance)) {
278             Relation relation = relationInstanceManager.getDatastoreType(instance);
279             sessionContext.getInstanceListenerService().preDelete(instance);
280             datastoreSession.getDatastoreRelationManager().deleteRelation(relation);
281             relationInstanceManager.removeInstance(instance);
282             relationInstanceManager.closeInstance(instance);
283             sessionContext.getInstanceListenerService().postDelete(instance);
284         } else {
285             throw new XOException(instance + " is not a managed XO instance.");
286         }

343     private TypeMetadataSet<EntityTypeMetadata<EntityMetadata>> getEffectiveTypes(Class<?> type, Class<?>... types) {
344         MetadataProvider<EntityMetadata, EntityDiscriminator, RelationMetadata, RelationDiscriminator> metadataProvider = sessionContext.getMetadataProvider();
345         TypeMetadataSet<EntityTypeMetadata<EntityMetadata>> effectiveTypes = new TypeMetadataSet<>();
