21  /**
22   * Allows resolving types from relation discriminators as provided by the datastores.
23   *
24   * @param <RelationDiscriminator> The discriminator type of the datastore (e.g. Neo4j relationship types or strings for JSON stores).

29      private final Map<RelationPropertyKey, AbstractRelationPropertyMethodMetadata<RelationMetadata>> relationProperties;

31      /**
32       * Constructor.
33       *
34       * @param metadataByType A map of all types with their metadata.
35       */
36      public RelationTypeMetadataResolver(Map<Class<?>, TypeMetadata> metadataByType) {
37          relationMappings = new HashMap<>();
38          relationProperties = new HashMap<>();
39          for (TypeMetadata typeMetadata : metadataByType.values()) {
40              if (typeMetadata instanceof RelationTypeMetadata) {
41                  RelationTypeMetadata<RelationMetadata> relationTypeMetadata = (RelationTypeMetadata) typeMetadata;
42                  Class<?> outgoingType = relationTypeMetadata.getFromType();
43                  EntityTypeMetadata<EntityMetadata> outgoingTypeMetadata = (EntityTypeMetadata<EntityMetadata>) metadataByType.get(outgoingType);
44                  Class<?> incomingType = relationTypeMetadata.getToType();
45                  EntityTypeMetadata<EntityMetadata> incomingTypeMetadata = (EntityTypeMetadata<EntityMetadata>) metadataByType.get(incomingType);
46                  RelationMapping<EntityDiscriminator, RelationMetadata, RelationDiscriminator> relationMapping = new RelationMapping<>(outgoingTypeMetadata.getDatastoreMetadata().getDiscriminator(), relationTypeMetadata, incomingTypeMetadata.getDatastoreMetadata().getDiscriminator());
47                  Set<RelationMapping<EntityDiscriminator, RelationMetadata, RelationDiscriminator>> mappingSet = relationMappings.get(relationTypeMetadata.getDatastoreMetadata().getDiscriminator());
48                  if (mappingSet == null) {
49                      mappingSet = new HashSet<>();
50                      relationMappings.put(relationTypeMetadata.getDatastoreMetadata().getDiscriminator(), mappingSet);
51                  }
52                  mappingSet.add(relationMapping);
53              } else if (typeMetadata instanceof EntityTypeMetadata) {
54                  EntityTypeMetadata<EntityMetadata> entityTypeMetadata = (EntityTypeMetadata<EntityMetadata>) typeMetadata;
55                  for (MethodMetadata<?, ?> methodMetadata : entityTypeMetadata.getProperties()) {
56                      if (methodMetadata instanceof AbstractRelationPropertyMethodMetadata<?>) {
57                          AbstractRelationPropertyMethodMetadata<RelationMetadata> propertyMethodMetadata = (AbstractRelationPropertyMethodMetadata<RelationMetadata>) methodMetadata;
58                          AnnotatedType relationType = propertyMethodMetadata.getRelationshipMetadata().getAnnotatedType();
59                          if (relationType != null) {
60                              Class<?> entityType = entityTypeMetadata.getAnnotatedType().getAnnotatedElement();
61                              RelationTypeMetadata<?> relationTypeMetadata = (RelationTypeMetadata<?>) metadataByType.get(relationType.getAnnotatedElement());
62                              Direction direction = propertyMethodMetadata.getDirection();
63                              relationProperties.put(new RelationPropertyKey(entityType, relationTypeMetadata, direction), propertyMethodMetadata);
64                          }
65                      }
66                  }
67              }
68          }
69      }

71      public TypeMetadataSet<RelationTypeMetadata<RelationMetadata>> getRelationTypes(Set<EntityDiscriminator> sourceDiscriminators, RelationDiscriminator discriminator, Set<EntityDiscriminator> targetDiscriminators) {
72          TypeMetadataSet<RelationTypeMetadata<RelationMetadata>> types = new TypeMetadataSet<>();
73          Set<RelationMapping<EntityDiscriminator, RelationMetadata, RelationDiscriminator>> relations = relationMappings.get(discriminator);
74          if (relations == null) {
75              throw new XOException("Cannot resolve relation from discriminator '" + discriminator + "'");
76          }
77          for (RelationMapping<EntityDiscriminator, RelationMetadata, RelationDiscriminator> relation : relations) {
78              EntityDiscriminator source = relation.getSource();
79              EntityDiscriminator target = relation.getTarget();
80              if (sourceDiscriminators.contains(source) && targetDiscriminators.contains(target)) {
81                  types.add(relation.getRelationType());
82              }
83          }
84          return types;
85      }

87      public AbstractRelationPropertyMethodMetadata<?> getRelationPropertyMethodMetadata(Class<?> type, RelationTypeMetadata<?> relationTypeMetadata, Direction direction) {
88          Class<?> containingType = null;
89          switch (direction) {
90              case FROM:
91                  if (relationTypeMetadata.getFromType().isAssignableFrom(type)) {
92                      containingType = relationTypeMetadata.getFromType();
93                  }
94                  break;
95              case TO:
96                  if (relationTypeMetadata.getToType().isAssignableFrom(type)) {
97                      containingType = relationTypeMetadata.getToType();
98                  }
99                  break;
100             default:
101                 throw direction.createNotSupportedException();
102         }
103         if (containingType == null) {
104             throw new XOException("Cannot resolve entity type containing a relation of type '" + relationTypeMetadata.getAnnotatedType().getName() + "'.");
105         }
106         RelationPropertyKey relationPropertyKey = new RelationPropertyKey(containingType, relationTypeMetadata, direction);
107         AbstractRelationPropertyMethodMetadata<?> propertyMethodMetadata = relationProperties.get(relationPropertyKey);
108         if (propertyMethodMetadata == null) {
109             throw new XOException("Cannot resolve property in type '" + containingType.getName() + "' for relation type '" + relationTypeMetadata.getAnnotatedType().getAnnotatedElement().getName() + "'.");
110         }
111         return propertyMethodMetadata;
112     }
