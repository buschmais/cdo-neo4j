35      public XOManagerFactoryImpl(XOUnit xoUnit) {
36          this.xoUnit = xoUnit;
37          Class<?> providerType = xoUnit.getProvider();
38          if (providerType == null) {
39              throw new XOException("No provider specified for XO unit '" + xoUnit.getName() + "'.");
40          }
41          if (!XODatastoreProvider.class.isAssignableFrom(providerType)) {
42              throw new XOException(providerType.getName() + " specified as XO provider must implement " + XODatastoreProvider.class.getName());
43          }
44          XODatastoreProvider<EntityMetadata, EntityDiscriminator, RelationMetadata, RelationDiscriminator> XODatastoreProvider = XODatastoreProvider.class.cast(ClassHelper.newInstance(providerType));
45          this.datastore = XODatastoreProvider.createDatastore(xoUnit);
46          this.pluginRepositoryManager = new PluginRepositoryManager(new QueryLanguagePluginRepository(datastore));
47          this.validationMode = xoUnit.getValidationMode();
48          this.concurrencyMode = xoUnit.getConcurrencyMode();
49          this.defaultTransactionAttribute = xoUnit.getDefaultTransactionAttribute();
50          final ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
51          final ClassLoader parentClassLoader = contextClassLoader != null ? contextClassLoader : xoUnit.getClass().getClassLoader();
52          LOGGER.debug("Using class loader '{}'.", parentClassLoader.toString());
53          classLoader = new ClassLoader() {
54              @Override
55              public Class<?> loadClass(String name) throws ClassNotFoundException {
56                  return parentClassLoader.loadClass(name);
57              }
58          };
59          metadataProvider = new MetadataProviderImpl(xoUnit.getTypes(), datastore);
60          this.validatorFactory = getValidatorFactory();
61          datastore.init(metadataProvider.getRegisteredMetadata());
62      }

82          SessionContext<EntityId, Entity, EntityMetadata, EntityDiscriminator, RelationId, Relation, RelationMetadata, RelationDiscriminator, PropertyMetadata> sessionContext = new SessionContext<>(metadataProvider, pluginRepositoryManager, datastoreSession, validatorFactory, xoUnit.getInstanceListeners(), defaultTransactionAttribute, validationMode, concurrencyMode, classLoader);
83          XOManagerImpl<EntityId, Entity, EntityMetadata, EntityDiscriminator, RelationId, Relation, RelationMetadata, RelationDiscriminator, PropertyMetadata> xoManager = new XOManagerImpl<>(sessionContext);
84          return sessionContext.getInterceptorFactory().addInterceptor(xoManager);
