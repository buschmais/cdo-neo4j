26      public RowProxyMethodService(SortedSet<Class<?>> types) {
27          BeanMethodProvider beanMethodProvider = BeanMethodProvider.newInstance();
28          for (Class<?> type : types) {
29              Collection<AnnotatedMethod> typeMethodsOfType = beanMethodProvider.getMethods(type);
30              for (AnnotatedMethod typeMethod : typeMethodsOfType) {
31                  if (!(typeMethod instanceof GetPropertyMethod)) {
32                      throw new XOException("Only get methods are supported for projections: '" + typeMethod.getAnnotatedElement().getName() + "'.");
33                  }
34                  PropertyMethod propertyMethod = (PropertyMethod) typeMethod;
35                  GetMethod proxyMethod = new GetMethod(propertyMethod.getName(), propertyMethod.getType());
36                  addProxyMethod(proxyMethod, propertyMethod.getAnnotatedElement());
37              }
38          }
39          addMethod(new AsMethod(), CompositeObject.class, "as", Class.class);
40          addMethod(new GetDelegateMethod<Map<String, Object>>(), CompositeObject.class, "getDelegate");
41          addMethod(new com.buschmais.xo.impl.proxy.query.row.GetMethod(), CompositeRowObject.class, "get", String.class, Class.class);
42          addMethod(new GetColumnsMethod(), CompositeRowObject.class, "getColumns");
43          addMethod(new HashCodeMethod(), Object.class, "hashCode");
44          addMethod(new EqualsMethod(), Object.class, "equals", Object.class);
45          addMethod(new ToStringMethod(), Object.class, "toString");
46      }
