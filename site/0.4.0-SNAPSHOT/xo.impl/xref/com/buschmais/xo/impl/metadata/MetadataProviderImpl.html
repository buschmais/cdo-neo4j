63      /**
64       * Constructor.
65       * 
66       * @param types
67       *            All classes as provided by the XO unit.
68       * @param datastore
69       *            The datastore.
70       */
71      public MetadataProviderImpl(Collection<Class<?>> types, Datastore<?, EntityMetadata, EntityDiscriminator, RelationMetadata, RelationDiscriminator> datastore) {
72          this.metadataFactory = datastore.getMetadataFactory();
73          DependencyResolver.DependencyProvider<Class<?>> classDependencyProvider = new DependencyResolver.DependencyProvider<Class<?>>() {
74              @Override
75              public Set<Class<?>> getDependencies(Class<?> dependent) {
76                  return new HashSet<>(Arrays.asList(dependent.getInterfaces()));
77              }
78          };
79          List<Class<?>> allClasses = DependencyResolver.newInstance(types, classDependencyProvider).resolve();
80          LOGGER.debug("Processing types {}", allClasses);
81          this.annotatedMethods = new HashMap<>();
82          for (Class<?> currentClass : allClasses) {
83              if (!currentClass.isInterface()) {
84                  throw new XOException("Type " + currentClass.getName() + " is not an interface.");
85              }
86              annotatedMethods.put(currentClass, BeanMethodProvider.newInstance().getMethods(currentClass));
87          }
88          for (Class<?> currentClass : allClasses) {
89              getOrCreateTypeMetadata(currentClass);
90          }
91          entityTypeMetadataResolver = new EntityTypeMetadataResolver<>(metadataByType);
92          relationTypeMetadataResolver = new RelationTypeMetadataResolver<>(metadataByType);
93          metadataByType.put(CompositeObject.class, new SimpleTypeMetadata(new AnnotatedType(CompositeObject.class), Collections.<TypeMetadata> emptyList(),
94                  Collections.<MethodMetadata<?, ?>> emptyList(), null));
95      }

169     /**
170      * Create the {@link TypeMetadata} for the given {@link AnnotatedType}.
171      * 
172      * @param annotatedType
173      *            The {@link AnnotatedType}.
174      * @return The corresponding metadata.
175      */
176     private TypeMetadata createTypeMetadata(AnnotatedType annotatedType) {
177         Class<?> currentClass = annotatedType.getAnnotatedElement();
178         List<TypeMetadata> superTypes = getSuperTypeMetadata(annotatedType);
179         Collection<AnnotatedMethod> annotatedMethods = this.annotatedMethods.get(currentClass);
180         Collection<MethodMetadata<?, ?>> methodMetadataOfType = getMethodMetadataOfType(annotatedType, annotatedMethods);
181         TypeMetadata metadata;
182         if (isEntityType(annotatedType)) {
183             metadata = createEntityTypeMetadata(annotatedType, superTypes, methodMetadataOfType);
184         } else if (isRelationType(annotatedType)) {
185             metadata = createRelationTypeMetadata(annotatedType, superTypes, methodMetadataOfType);
186         } else {
187             IndexedPropertyMethodMetadata indexedProperty = getIndexedPropertyMethodMetadata(methodMetadataOfType);
188             metadata = new SimpleTypeMetadata(annotatedType, superTypes, methodMetadataOfType, indexedProperty);
189         }
190         return metadata;
191     }

216     /**
217      * Determines if an {@link AnnotatedType} represents a specific type
218      * identified by a meta annotation.
219      * 
220      * @param annotatedType
221      *            The {@link AnnotatedType}.
222      * @param definitionType
223      *            The meta annotation.
224      * @return <code>true</code> if the annotated type represents relation type.
225      */
226     private boolean isOfDefinitionType(AnnotatedType annotatedType, Class<? extends Annotation> definitionType) {
227         Annotation definition = annotatedType.getByMetaAnnotation(definitionType);
228         if (definition != null) {
229             return true;
230         }
231         for (Class<?> superType : annotatedType.getAnnotatedElement().getInterfaces()) {
232             if (isOfDefinitionType(new AnnotatedType(superType), definitionType)) {
233                 return true;
234             }
235         }
236         return false;
237     }

259     /**
260      * Get or create a {@link RelationTypeMetadata} instance for the given
261      * {@link AnnotatedType}.
262      * 
263      * @param annotatedType
264      *            The {@link AnnotatedType}.
265      * @param superTypes
266      *            The metadata collection of the super types.
267      * @param methodMetadataOfType
268      *            The method metadata of the type.
269      * @return The {@link RelationTypeMetadata} instance representing the
270      *         annotated type.
271      */
272     private RelationTypeMetadata<RelationMetadata> createRelationTypeMetadata(AnnotatedType annotatedType, List<TypeMetadata> superTypes,
273             Collection<MethodMetadata<?, ?>> methodMetadataOfType) {
274         Class<?> fromType = null;
275         Class<?> toType = null;
276         Collection<MethodMetadata<?, ?>> current = methodMetadataOfType;
277         Queue<TypeMetadata> queue = new LinkedList<>(superTypes);
278         // Starting from the type to be created search all its properties and
279         // those of its super types for reference properties defining the from
280         // and to entity types
281         do {
282             for (MethodMetadata<?, ?> methodMetadata : current) {
283                 if (methodMetadata instanceof EntityReferencePropertyMethodMetadata) {
284                     EntityReferencePropertyMethodMetadata<?> propertyMethodMetadata = (EntityReferencePropertyMethodMetadata<?>) methodMetadata;
285                     Class<?> type = propertyMethodMetadata.getAnnotatedMethod().getType();
286                     switch (propertyMethodMetadata.getDirection()) {
287                     case FROM:
288                         fromType = type;
289                         break;
290                     case TO:
291                         toType = type;
292                         break;
293                     default:
294                         throw propertyMethodMetadata.getDirection().createNotSupportedException();
295                     }
296                 }
297             }
298             TypeMetadata superType = queue.poll();
299             if (superType != null) {
300                 queue.addAll(superType.getSuperTypes());
301                 current = superType.getProperties();
302             } else {
303                 current = null;
304             }
305         } while (current != null && fromType == null && toType == null);
306         if (fromType == null || toType == null) {
307             throw new XOException("Relation type '" + annotatedType.getAnnotatedElement().getName()
308                     + "' does not define target entity properties for both directions.");
309         }
310         RelationMetadata relationMetadata = metadataFactory.createRelationMetadata(annotatedType, metadataByType);
311         RelationTypeMetadata<RelationMetadata> relationTypeMetadata = new RelationTypeMetadata<>(annotatedType, superTypes, methodMetadataOfType, fromType,
312                 toType, relationMetadata);
313         metadataByType.put(annotatedType.getAnnotatedElement(), relationTypeMetadata);
314         return relationTypeMetadata;
315     }

333     /**
334      * Determine the indexed property from a list of method metadata.
335      * 
336      * @param methodMetadataOfType
337      *            The list of method metadata.
338      * @return The {@link IndexedPropertyMethodMetadata}.
339      */
340     private IndexedPropertyMethodMetadata<?> getIndexedPropertyMethodMetadata(Collection<MethodMetadata<?, ?>> methodMetadataOfType) {
341         IndexedPropertyMethodMetadata indexedProperty = null;
342         for (MethodMetadata methodMetadata : methodMetadataOfType) {
343             AnnotatedMethod annotatedMethod = methodMetadata.getAnnotatedMethod();
344             Annotation indexedAnnotation = annotatedMethod.getByMetaAnnotation(IndexDefinition.class);
345             if (indexedAnnotation != null) {
346                 if (!(methodMetadata instanceof PrimitivePropertyMethodMetadata)) {
347                     throw new XOException("Only primitive properties are allowed to be used for indexing.");
348                 }
349                 indexedProperty = new IndexedPropertyMethodMetadata<>((PropertyMethod) annotatedMethod, (PrimitivePropertyMethodMetadata) methodMetadata,
350                         metadataFactory.createIndexedPropertyMetadata((PropertyMethod) annotatedMethod));
351             }
352         }
353         return indexedProperty;
354     }

356     /**
357      * Return the collection of method metadata from the given collection of
358      * annotateed methods.
359      * 
360      * @param annotatedMethods
361      *            The collection of annotated methods.
362      * @return The collection of method metadata.
363      */
364     private Collection<MethodMetadata<?, ?>> getMethodMetadataOfType(AnnotatedType annotatedType, Collection<AnnotatedMethod> annotatedMethods) {
365         Collection<MethodMetadata<?, ?>> methodMetadataOfType = new ArrayList<>();
366         // Collect the getter methods as they provide annotations holding meta
367         // information also to be applied to setters
368         for (AnnotatedMethod annotatedMethod : annotatedMethods) {
369             MethodMetadata<?, ?> methodMetadata;
370             ImplementedBy implementedBy = annotatedMethod.getAnnotation(ImplementedBy.class);
371             ResultOf resultOf = annotatedMethod.getAnnotation(ResultOf.class);
372             if (implementedBy != null) {
373                 methodMetadata = new ImplementedByMethodMetadata<>(annotatedMethod, implementedBy.value(),
374                         metadataFactory.createImplementedByMetadata(annotatedMethod));
375             } else if (resultOf != null) {
376                 methodMetadata = createResultOfMetadata(annotatedMethod, resultOf);
377             } else if (annotatedMethod instanceof PropertyMethod) {
378                 PropertyMethod propertyMethod = (PropertyMethod) annotatedMethod;
379                 Transient transientAnnotation = propertyMethod.getAnnotationOfProperty(Transient.class);
380                 if (transientAnnotation != null) {
381                     methodMetadata = new TransientPropertyMethodMetadata(propertyMethod);
382                 } else
383                     methodMetadata = createPropertyMethodMetadata(annotatedType, propertyMethod);
384             } else {
385                 methodMetadata = new UnsupportedOperationMethodMetadata((UserMethod) annotatedMethod);
386             }
387             methodMetadataOfType.add(methodMetadata);
388         }
389         return methodMetadataOfType;
390     }

392     /**
393      * Create the {@link MethodMetadata} for a property method.
394      * 
395      * @param annotatedType
396      *            The annotated type containing the property.
397      * @param propertyMethod
398      *            The property method.
399      * @return The {@link MethodMetadata}.
400      */
401     private MethodMetadata<?, ?> createPropertyMethodMetadata(AnnotatedType annotatedType, PropertyMethod propertyMethod) {
402         MethodMetadata<?, ?> methodMetadata;
403         Class<?> propertyType = propertyMethod.getType();
404         if (Collection.class.isAssignableFrom(propertyType)) {
405             Type genericType = propertyMethod.getGenericType();
406             ParameterizedType type = (ParameterizedType) genericType;
407             Class<?> elementType;
408             Type typeArgument = type.getActualTypeArguments()[0];
409             if (typeArgument instanceof Class) {
410                 elementType = (Class<?>) typeArgument;
411             } else if (typeArgument instanceof ParameterizedType) {
412                 ParameterizedType parameterizedTypeArgument = (ParameterizedType) typeArgument;
413                 elementType = (Class<?>) parameterizedTypeArgument.getRawType();
414             } else {
415                 throw new XOException("Cannot determine argument type of collection property " + propertyMethod.getAnnotatedElement().toGenericString());
416             }
417             AnnotatedType annotatedTypeArgument = new AnnotatedType(elementType);
418             if (isEntityType(annotatedTypeArgument)) {
419                 Direction relationDirection = getRelationDirection(propertyMethod, Direction.FROM);
420                 com.buschmais.xo.spi.reflection.AnnotatedElement<?> relationElement = getRelationDefinitionElement(propertyMethod);
421                 RelationTypeMetadata relationshipType = new RelationTypeMetadata<>(metadataFactory.createRelationMetadata(relationElement, metadataByType));
422                 methodMetadata = new EntityCollectionPropertyMethodMetadata<>(propertyMethod, relationshipType, relationDirection,
423                         metadataFactory.createCollectionPropertyMetadata(propertyMethod));
424             } else if (isRelationType(annotatedTypeArgument)) {
425                 TypeMetadata relationTypeMetadata = getOrCreateTypeMetadata(elementType);
426                 RelationTypeMetadata<RelationMetadata> relationMetadata = (RelationTypeMetadata) relationTypeMetadata;
427                 Direction relationDirection = getRelationDirection(annotatedType, propertyMethod, relationTypeMetadata, relationMetadata);
428                 methodMetadata = new RelationCollectionPropertyMethodMetadata<>(propertyMethod, relationMetadata, relationDirection,
429                         metadataFactory.createCollectionPropertyMetadata(propertyMethod));
430             } else {
431                 throw new XOException("Unsupported type argument '" + elementType.getName() + "' for collection property: " + propertyType.getName());
432             }
433         } else if (annotatedMethods.containsKey(propertyType)) {
434             AnnotatedType referencedType = new AnnotatedType(propertyType);
435             Direction relationDirection;
436             RelationTypeMetadata<RelationMetadata> relationMetadata;
437             if (isEntityType(referencedType)) {
438                 relationDirection = getRelationDirection(propertyMethod, Direction.FROM);
439                 com.buschmais.xo.spi.reflection.AnnotatedElement<?> relationElement = getRelationDefinitionElement(propertyMethod);
440                 relationMetadata = new RelationTypeMetadata<>(metadataFactory.createRelationMetadata(relationElement, metadataByType));
441                 methodMetadata = new EntityReferencePropertyMethodMetadata<>(propertyMethod, relationMetadata, relationDirection,
442                         metadataFactory.createReferencePropertyMetadata(propertyMethod));
443             } else if (isRelationType(referencedType)) {
444                 TypeMetadata relationTypeMetadata = getOrCreateTypeMetadata(propertyType);
445                 relationMetadata = (RelationTypeMetadata) relationTypeMetadata;
446                 relationDirection = getRelationDirection(annotatedType, propertyMethod, relationTypeMetadata, relationMetadata);
447                 methodMetadata = new RelationReferencePropertyMethodMetadata<>(propertyMethod, relationMetadata, relationDirection,
448                         metadataFactory.createReferencePropertyMetadata(propertyMethod));
449             } else {
450                 throw new XOException("Unsupported type for reference property: " + propertyType.getName());
451             }
452         } else {
453             methodMetadata = new PrimitivePropertyMethodMetadata<>(propertyMethod, metadataFactory.createPropertyMetadata(propertyMethod));
454         }
455         return methodMetadata;
456     }

458     private Direction getRelationDirection(PropertyMethod propertyMethod, Direction defaultDirection) {
459         Annotation fromAnnotation = propertyMethod.getByMetaAnnotationOfProperty(FromDefinition.class);
460         Annotation toAnnotation = propertyMethod.getByMetaAnnotationOfProperty(ToDefinition.class);
461         if (fromAnnotation != null && toAnnotation != null) {
462             throw new XOException("The relation property '" + propertyMethod.getName() + "' must not specifiy both directions.'");
463         }
464         Direction direction = null;
465         if (fromAnnotation != null) {
466             direction = RelationTypeMetadata.Direction.FROM;
467         }
468         if (toAnnotation != null) {
469             direction = RelationTypeMetadata.Direction.TO;
470         }
471         return direction != null ? direction : defaultDirection;
472     }

474     private Direction getRelationDirection(AnnotatedType annotatedEntityType, PropertyMethod propertyMethod, TypeMetadata propertyTypeMetadata,
475             RelationTypeMetadata<RelationMetadata> relationMetadata) {
476         Direction relationDirection = getRelationDirection(propertyMethod, null);
477         if (relationDirection == null) {
478             Class<?> fromType = relationMetadata.getFromType();
479             Class<?> toType = relationMetadata.getToType();
480             if (fromType.equals(toType)) {
481                 throw new XOException("Direction of property '" + propertyMethod.getAnnotatedElement().toGenericString()
482                         + "' is ambiguous and must be specified.");
483             }
484             if (annotatedEntityType.getAnnotatedElement().equals(fromType)) {
485                 relationDirection = Direction.FROM;
486             } else if (annotatedEntityType.getAnnotatedElement().equals(toType)) {
487                 relationDirection = Direction.TO;
488             } else {
489                 throw new XOException("Cannot determine relation direction for type '" + propertyTypeMetadata.getAnnotatedType().getName() + "'");
490             }
491         }
492         return relationDirection;
493     }

495     /**
496      * Determines the {@link AnnotatedElement} which is annotated with
497      * {@link RelationDefinition}.
498      * 
499      * @param propertyMethod
500      *            The property method to start with.
501      * @return The annotated element.
502      */
503     private com.buschmais.xo.spi.reflection.AnnotatedElement<?> getRelationDefinitionElement(PropertyMethod propertyMethod) {
504         if (propertyMethod.getByMetaAnnotationOfProperty(RelationDefinition.class) != null) {
505             return propertyMethod;
506         }
507         Annotation[] declaredAnnotations = propertyMethod.getAnnotationsOfProperty();
508         for (int i = 0; i < declaredAnnotations.length; i++) {
509             com.buschmais.xo.spi.reflection.AnnotatedElement<?> annotationTypeElement = new AnnotatedType(declaredAnnotations[i].annotationType());
510             if (annotationTypeElement.getByMetaAnnotation(RelationDefinition.class) != null) {
511                 return annotationTypeElement;
512             }
513         }
514         return propertyMethod;
515     }

517     /**
518      * Creates the method metadata for methods annotated with {@link ResultOf}.
519      * 
520      * @param annotatedMethod
521      *            The annotated method-
522      * @param resultOf
523      *            The {@link com.buschmais.xo.api.annotation.ResultOf}
524      *            annotation.
525      * @return The method metadata.
526      */
527     private MethodMetadata<?, ?> createResultOfMetadata(AnnotatedMethod annotatedMethod, ResultOf resultOf) {
528         Method method = annotatedMethod.getAnnotatedElement();
529 
530         // Determine query type
531         Class<?> methodReturnType = method.getReturnType();
532         Class<?> returnType;
533         if (Result.class.isAssignableFrom(methodReturnType)) {
534             Type genericReturnType = method.getGenericReturnType();
535             ParameterizedType parameterizedType = (ParameterizedType) genericReturnType;
536             returnType = (Class<?>) parameterizedType.getActualTypeArguments()[0];
537         } else {
538             returnType = methodReturnType;
539         }
540 
541         // Determine query type
542         AnnotatedElement query = resultOf.query();
543         if (Object.class.equals(query)) {
544             if (annotatedMethod.getByMetaAnnotation(QueryDefinition.class) != null) {
545                 query = annotatedMethod.getAnnotatedElement();
546             } else {
547                 query = returnType;
548             }
549         }
550         // Determine parameter bindings
551         Annotation[][] parameterAnnotations = method.getParameterAnnotations();
552         List<Parameter> parameters = new ArrayList<>();
553         for (Annotation[] parameterAnnotation : parameterAnnotations) {
554             Parameter parameter = null;
555             for (Annotation annotation : parameterAnnotation) {
556                 if (Parameter.class.equals(annotation.annotationType())) {
557                     parameter = (Parameter) annotation;
558                 }
559             }
560             if (parameter == null) {
561                 throw new XOException("Cannot determine parameter names for '" + method.getName() + "', all parameters must be annotated with '"
562                         + Parameter.class.getName() + "'.");
563             }
564             parameters.add(parameter);
565         }
566         boolean singleResult = !Iterable.class.isAssignableFrom(methodReturnType);
567         return new ResultOfMethodMetadata<>(annotatedMethod, query, returnType, resultOf.usingThisAs(), parameters, singleResult);
568     }
