43      public Set<String> getEntityDiscriminators(ObjectNode jsonNodes) {
44          ArrayNode typesNode = (ArrayNode) jsonNodes.get(TYPES_PROPERTY);
45          Set<String> discriminators = new HashSet<>();

58      @Override
59      public ObjectNode createEntity(TypeMetadataSet<EntityTypeMetadata<JsonNodeMetadata>> types, Set<String> discriminators,Map<PrimitivePropertyMethodMetadata<JsonPropertyMetadata>, Object> exampleEntity) {
60          ObjectNode rootNode = mapper.createObjectNode();
61          ArrayNode typesNode = mapper.createArrayNode();
62          for (String typeName : discriminators) {
63              typesNode.add(typeName);
64          }
65          rootNode.put(TYPES_PROPERTY, typesNode);
66          UUID uuid = UUID.randomUUID();
67          rootNode.put(ID_PROPERTY, uuid.toString());
68          for (Map.Entry<PrimitivePropertyMethodMetadata<JsonPropertyMetadata>, Object> entry : exampleEntity.entrySet()) {
69              setProperty(rootNode, entry.getKey(), entry.getValue());
70          }
71          return rootNode;
72      }

103     public void setProperty(ObjectNode objectNode, PrimitivePropertyMethodMetadata<JsonPropertyMetadata> metadata, Object value) {
104         Class<?> type = metadata.getAnnotatedMethod().getType();
105         if (String.class.equals(type)) {
106             objectNode.put(metadata.getAnnotatedMethod().getName(), (String) value);
107         } else {
108             throw new XOException("Unsupported type " + type + " for property " + metadata.getAnnotatedMethod().getName());
109         }
110     }
